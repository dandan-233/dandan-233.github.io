# 将本地文件夹关联到git仓库

# 创建git仓库

直接在网页或者其他客户端创建需要关联的仓库

# 本地文件夹关联

在文件夹根目录下执行

# 初始化 git

git init

# 将要提交的文件提交到暂存区

git add .

# 提交文件添加备注

git commit -m '备注描述内容'

# 关联远端仓库

git remote add origin git@github.com:用户名/仓库名.git

# 注意: 如果远端仓库有内容则先pull,再push

git pull -rebase origin master

# 推送文件到远端

git push -u origin master
